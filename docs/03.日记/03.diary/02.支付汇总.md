## 一.微信h5端

### 1.微信扫码：主要是通过订单id和openid获取二维码的base64地址

### 2.支付宝扫码：需要订单id

### 3.微信拉起支付：需要配置jweixin.config，通过jweixin.invoke或者wx的chooseWxpay。

##### 应注意: 必须配置jweixin.conifg且jsAPiList中必须包含对应的接口名称，支付必须要求openid，所以在之前必须进行静默或费静默的用户授权来获得code，再解析code获取openid。（仅支持线上模式）

###### 代码:

```vue
<template>
  <view class="page">
    <u-no-network></u-no-network>
    <u-toast ref="uToast" />

      <u-loading mode="circle" size="100" color="RED"></u-loading>
    </view>

      </view>

        </view>

      </view>

<script>
  import api from "@/common/api";
  import {
    APPINFO
  } from "@/common/constant.js"
  import {
    wxApi
  } from '@/common/wxApi.js'
  import jweixin from 'jweixin-module'

  export default {
    data() {
      return {
        loading: false,
        hash_id: "",
        price: "",
        disabled: true,
        shareid: '',
        timer: null,
        openid: '',
        pay_status: 0
      };
    },
    onShow() {
      // #ifdef H5
      if (this.$wechat && this.$wechat.isWechat()) {
        let _self = this
        this.timer = setTimeout(()=>{
          _self.disabled = false
        }, 800)
      }
      // #endif
    },
    onLoad() {
      this.price = uni.getStorageSync('price')
      this.hash_id = uni.getStorageSync('order_hash_id')
      this.openid = uni.getStorageSync('openid')

    },
    onUnload() {
      clearTimeout(this.timer)
      this.timer = null
    },
    methods: {
      // #ifdef H5
      // 注册wxjsskd
      async registerWxConfig() {
        let request = await api.getWx_confing({
          query: {
            url: window.location.href,
            jsApiList: "chooseWXPay"
          }
        });
        if (request.statusCode === 200) {
          let wx_config = request.data.data;
          //初始化sdk配置
          jweixin.config({
            debug: false,
            appId: APPINFO.appid,
            jsApiList: ['chooseWXPay'],
            signature: wx_config.signature,
            timestamp: wx_config.timestamp,
            nonceStr: wx_config.nonceStr
          });
        }
      },
      async getToken(code) {
        let request = await api.getToken({
          query: {
            code: code
          },
          method: 'POST'
        });
        if (request.statusCode === 200) {
          this.openid = request.data.data.openid
        } else {
          this.$refs.uTips.show({
            title: request.data.message,
            type: "warning",
            duration: 1500
          })
          setTimeout(function() {
            uni.navigateTo({
              url: "/pages/pay/online"
            })
          }, 1500);
        }
      },
      async confirm() {
        this.disabled = true;
        let _self = this;
        let params = {
          query: {
            order_hash_id: _self.hash_id,
            openid: _self.openid
          },
          method: "POST"
        }
        let request = await api.pay(params);
        jweixin.chooseWXPay({
            appId: request.data.data.appId,
            nonceStr: request.data.data.nonceStr,
            package: request.data.data.package,
            paySign: request.data.data.paySign,
            signType: request.data.data.signType,
            timestamp: request.data.data.timeStamp,
            success: function (res) {
              // 支付成功后的回调函数
              console.log(res)
              uni.redirectTo({
                // url: "/pages/index/detail?hash_id="+_self.hash_id
                url: "/pages/pay/success"
              });
              _self.disabled = false;
            }
        })
      },
      // #endif
    }
  };
</script>
```

## 二.微信小程序支付

### 1.微信小程序jsapi的支付需要在微信商户中配置（需要先拥有商户才能使用）

### 2.在产品中心中进行域名授权（需注意第一次添加域名授权的时候回跳转至操作密码验证，验证完毕后才能进行真正的域名授权)

### 3.配置api秘钥和下载api证书(证书放在服务器上)还需要appid和商户号，这部分是后端相关

### 4.前端代码编写:

```js
let a = await api.getOrderInfo()	// 返回相关配置包括订单号
uni.requestPayment({...a, provider: "weixin"})	// provider也可以通过getProvider获得详见uni官网
```

### 5.以下是详细的支付代码(uni-app平台)包括app端，来自github的hello-uniapp项目

```vue
<template>
	<view>
		<page-head :title="title"></page-head>
		<view class="uni-padding-wrap">
			<view style="background:#FFF; padding:50rpx 0;">
				<view class="uni-hello-text uni-center"><text>支付金额</text></view>
				<view class="uni-h1 uni-center uni-common-mt">
					<text class="rmbLogo">￥</text>
					<input class="price" type="digit" :value="price" maxlength="4" @input="priceChange" />
				</view>
			</view>
			<view class="uni-btn-v uni-common-mt">
				<!-- #ifdef MP-WEIXIN -->
				<button type="primary" @click="weixinPay" :loading="loading">微信支付</button>
				<!-- #endif -->
				<!-- #ifdef APP-PLUS -->
				<template v-if="providerList.length > 0">
					<button v-for="(item,index) in providerList" :key="index" @click="requestPayment(item,index)" :loading="item.loading">{{item.name}}支付</button>
				</template>
				<!-- #endif -->
			</view>
		</view>
	</view>
</template>
<script>
	export default {
		data() {
			return {
				title: 'request-payment',
				loading: false,
				price: 1,
				providerList: []
			}
		},
		onLoad: function() {
			// #ifdef APP-PLUS
			uni.getProvider({
				service: "payment",
				success: (e) => {
					console.log("payment success:" + JSON.stringify(e));
					let providerList = [];
					e.provider.map((value) => {
						switch (value) {
							case 'alipay':
								providerList.push({
									name: '支付宝',
									id: value,
									loading: false
								});
								break;
							case 'wxpay':
								providerList.push({
									name: '微信',
									id: value,
									loading: false
								});
								break;
							default:
								break;
						}
					})
					this.providerList = providerList;
				},
				fail: (e) => {
					console.log("获取支付通道失败：", e);
				}
			});
			// #endif
		},
		methods: {
			loginMpWeixin() {
				return new Promise((resolve, reject) => {
					uni.login({
						provider: 'weixin',
						success(res) {
							uni.request({
								url: 'https://97fca9f2-41f6-449f-a35e-3f135d4c3875.bspapp.com/http/user-center',
								method: 'POST',
								data: {
									action: 'loginByWeixin',
									params: {
										code: res.code,
										platform: 'mp-weixin'
									}
								},
								success(res) {
									if (res.data.code !== 0) {
										reject(new Error('获取openid失败：', res.result.msg))
										return
									}
									uni.setStorageSync('openid', res.data.openid)
									resolve(res.data.openid)
								},
								fail(err) {
									reject(new Error('获取openid失败：' + err))
								}
							})
						},
						fail(err) {
							reject(err)
						}
					})
				})
			},
			async weixinPay() {
				let openid = uni.getStorageSync('openid')
				console.log("发起支付");
				this.loading = true;
				if (!openid) {
					try {
						openid = await this.loginMpWeixin()
					} catch (e) {
						console.error(e)
					}
					if (!openid) {
						uni.showModal({
							content: '获取openid失败',
							showCancel: false
						})
						this.loading = false
						return
					}
				}
				this.openid = openid
				let orderInfo = await this.getOrderInfo('wxpay')
				if (!orderInfo) {
					uni.showModal({
						content: '获取支付信息失败',
						showCancel: false
					})
					return
				}
				uni.requestPayment({
					...orderInfo,
					success: (res) => {
						uni.showToast({
							title: "感谢您的赞助!"
						})
					},
					fail: (res) => {
						uni.showModal({
							content: "支付失败,原因为: " + res
								.errMsg,
							showCancel: false
						})
					},
					complete: () => {
						this.loading = false;
					}
				})
			},
			async requestPayment(e, index) {
				this.providerList[index].loading = true;
				const provider = e.id
				let orderInfo = await this.getOrderInfo(provider);
				if (!orderInfo) {
					uni.showModal({
						content: '获取支付信息失败',
						showCancel: false
					})
					return
				}
				console.log('--------orderInfo--------')
				console.log(orderInfo);
				uni.requestPayment({
					provider,
					orderInfo: orderInfo,
					success: (e) => {
						console.log("success", e);
						uni.showToast({
							title: "感谢您的赞助!"
						})
					},
					fail: (e) => {
						console.log("fail", e);
						uni.showModal({
							content: "支付失败,原因为: " + e.errMsg,
							showCancel: false
						})
					},
					complete: () => {
						this.providerList[index].loading = false;
					}
				})
			},
			getOrderInfo(provider) {
				return new Promise((resolve, reject) => {
					if (!this.price) {
						reject(new Error('请输入金额'))
					}
					uni.request({
						method: 'POST',
						url: 'https://97fca9f2-41f6-449f-a35e-3f135d4c3875.bspapp.com/http/pay',
						data: {
							provider,
							openid: this.openid,
							totalFee: Number(this.price) * 100, // 转为以分为单位
							// #ifdef APP-PLUS
							platform: 'app-plus',
							// #endif
							// #ifdef MP-WEIXIN
							platform: 'mp-weixin',
							// #endif
						},
						success(res) {
							if (res.data.code === 0) {
								resolve(res.data.orderInfo)
							} else {
								reject(new Error('获取支付信息失败' + res.data.msg))
							}
						},
						fail(err) {
							reject(new Error('请求支付接口失败' + err))
						}
					})
				})
			},
			priceChange(e) {
				console.log(e.detail.value)
				this.price = e.detail.value;
			}
		}
	}
</script>
```
